let totalPoints = 0; window.onload = function () { loadMiles(); }; function loadMiles() { fetch(`${API}/loyalty.php?action=history`, { credentials: 'include' }) .then(res => res.json()) .then(data => { if (!data.success) { window.location.href = 'login.html'; return; } const history = data.data.history || []; const earned = data.data.earned || 0; const redeemed= data.data.redeemed|| 0; totalPoints = data.data.balance || 0; if(el('hero-points')) el('hero-points').textContent = totalPoints.toLocaleString(); if(el('total-earned')) el('total-earned').textContent = earned.toLocaleString(); if(el('total-redeemed')) el('total-redeemed').textContent = redeemed.toLocaleString(); if(el('current-balance')) el('current-balance').textContent = totalPoints.toLocaleString(); updateTier(totalPoints); renderHistory(history); }) .catch(() => window.location.href = 'login.html'); } function updateTier(points) { let tier, next, nextPoints, pct; if (points >= 50000) { tier='Platinum'; next='Maximum tier!'; nextPoints=50000; pct=100; } else if (points >= 20000) { tier='Gold'; next=`${(50000-points).toLocaleString()} miles to Platinum`; nextPoints=50000; pct=Math.round((points/50000)*100); } else if (points >= 5000) { tier='Silver'; next=`${(20000-points).toLocaleString()} miles to Gold`; nextPoints=20000; pct=Math.round((points/20000)*100); } else { tier='Bronze'; next=`${(5000-points).toLocaleString()} miles to Silver`; nextPoints=5000; pct=Math.round((points/5000)*100); } const badge = el('tier-badge'); if (badge) { badge.textContent = tier; badge.className = 'tier-badge tier-' + tier.toLowerCase(); } if(el('tier-next')) el('tier-next').textContent = next; if(el('progress-label')) el('progress-label').textContent = `${points.toLocaleString()} / ${nextPoints.toLocaleString()} miles`; if(el('progress-fill')) el('progress-fill').style.width = pct + '%'; } function renderHistory(history) { const list = el('txn-list'); if (!list) return; if (history.length === 0) { list.innerHTML = '

No miles history yet. Book a flight to start earning!

'; return; } list.innerHTML = history.map(t => `
${t.description}
${t.created_at}
${t.transaction_type === 'Earn' ? '+' : '-'}${Number(t.points).toLocaleString()} miles
`).join(''); } function redeemReward(reward, cost) { if (totalPoints < cost) { alert(`You need ${cost.toLocaleString()} miles. You have ${totalPoints.toLocaleString()}.`); return; } if (!confirm(`Redeem ${cost.toLocaleString()} miles?`)) return; fetch(`${API}/loyalty.php?action=redeem`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, credentials: 'include', body: JSON.stringify({ reward, cost }) }) .then(res => res.json()) .then(data => { if (data.success) { alert('✅ Reward redeemed!'); loadMiles(); } else { alert(data.message); } }); } function logoutUser() { fetch(`${API}/auth.php?action=logout`, { credentials: 'include' }) .then(() => window.location.href = 'login.html'); }
