// ── PAYMENT PAGE ── let paymentData = {}; let selectedMethod = 'card'; window.onload = function () { const p = new URLSearchParams(window.location.search); paymentData = { booking_ref: p.get('booking_ref') || '', seat: p.get('seat') || '', flight_id: p.get('flight_id') || '', total: parseFloat(p.get('total')) || 0 }; if (!paymentData.booking_ref) { window.location.href = 'flights.html'; return; } displaySummary(); }; function displaySummary() { const tax = Math.round(paymentData.total * 0.05); const grand = paymentData.total + tax; if (el('disp-ref')) el('disp-ref').textContent = paymentData.booking_ref; if (el('disp-seat')) el('disp-seat').textContent = paymentData.seat || '—'; if (el('disp-flight')) el('disp-flight').textContent = 'Flight #' + paymentData.flight_id; if (el('disp-amount')) el('disp-amount').textContent = 'PKR ' + grand.toLocaleString(); if (el('pb-base')) el('pb-base').textContent = 'PKR ' + paymentData.total.toLocaleString(); if (el('pb-tax')) el('pb-tax').textContent = 'PKR ' + tax.toLocaleString(); if (el('pb-total')) el('pb-total').textContent = 'PKR ' + grand.toLocaleString(); } function selectMethod(method) { selectedMethod = method; document.querySelectorAll('.pay-method').forEach(m => m.classList.remove('selected')); event.currentTarget.classList.add('selected'); document.querySelectorAll('.method-panel').forEach(p => p.classList.remove('active')); const panel = document.getElementById('panel-' + method); if (panel) panel.classList.add('active'); } function formatCard(input) { let val = input.value.replace(/\D/g, '').substring(0, 16); input.value = val.replace(/(.{4})/g, '$1 ').trim(); } function formatExpiry(input) { let val = input.value.replace(/\D/g, '').substring(0, 4); if (val.length >= 2) val = val.substring(0,2) + '/' + val.substring(2); input.value = val; } function validatePayment() { if (selectedMethod === 'card') { const name = el('card-name') ? el('card-name').value.trim() : ''; const number = el('card-number') ? el('card-number').value.replace(/\s/g,'') : ''; const expiry = el('card-expiry') ? el('card-expiry').value.trim() : ''; const cvv = el('card-cvv') ? el('card-cvv').value.trim() : ''; if (!name) { alert('Please enter cardholder name.'); return false; } if (number.length < 16) { alert('Please enter a valid card number.'); return false; } if (expiry.length < 5) { alert('Please enter a valid expiry date.'); return false; } if (cvv.length < 3) { alert('Please enter a valid CVV.'); return false; } } if (selectedMethod === 'jazzcash') { const num = el('jazz-number') ? el('jazz-number').value.trim() : ''; const mpin = el('jazz-mpin') ? el('jazz-mpin').value.trim() : ''; if (num.length < 11) { alert('Please enter a valid JazzCash number.'); return false; } if (mpin.length < 4) { alert('Please enter your JazzCash MPIN.'); return false; } } if (selectedMethod === 'easypaisa') { const num = el('easy-number') ? el('easy-number').value.trim() : ''; const mpin = el('easy-mpin') ? el('easy-mpin').value.trim() : ''; if (num.length < 11) { alert('Please enter a valid Easypaisa number.'); return false; } if (mpin.length < 4) { alert('Please enter your Easypaisa MPIN.'); return false; } } return true; } function processPayment() { if (!validatePayment()) return; const tax = Math.round(paymentData.total * 0.05); const grand = paymentData.total + tax; const btn = document.querySelector('.btn-pay'); btn.disabled = true; btn.textContent = 'Processing...'; fetch(`${API}/payment.php?action=pay`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, credentials: 'include', body: JSON.stringify({ booking_ref: paymentData.booking_ref, seat: paymentData.seat, amount: grand, payment_method: selectedMethod }) }) .then(res => res.json()) .then(data => { if (data.success) { window.location.href = `boarding-pass.html?booking_ref=${paymentData.booking_ref}&seat=${paymentData.seat}`; } else { alert(data.message); btn.disabled = false; btn.textContent = '🔒 Pay Now'; } }) .catch(() => { alert('Connection error. Make sure XAMPP is running.'); btn.disabled = false; btn.textContent = '🔒 Pay Now'; }); }
