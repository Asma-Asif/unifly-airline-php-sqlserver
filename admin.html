// ── FLIGHTS PAGE ── function getParams() { const p = new URLSearchParams(window.location.search); return { from: p.get('from') || '', to: p.get('to') || '', date: p.get('date') || '', cls: p.get('class') || 'Economy' }; } window.onload = function () { const params = getParams(); if (el('from')) el('from').value = params.from; if (el('to')) el('to').value = params.to; if (el('date')) el('date').value = params.date; if (el('class')) el('class').value = params.cls; if (params.from && params.to && params.date) { loadFlights(params.from, params.to, params.date, params.cls); } }; function loadFlights(from, to, date, cls) { const info = el('results-info'); const list = el('flights-list'); if (!info || !list) return; info.innerHTML = '

Searching flights...

'; list.innerHTML = ''; fetch(`${API}/flights.php?action=search&from=${from}&to=${to}&date=${date}&class=${cls}`) .then(res => res.json()) .then(data => { if (!data.success) { info.innerHTML = `

${data.message}

`; return; } const flights = data.data.flights; if (!flights || flights.length === 0) { info.innerHTML = '

No flights found for this route and date.

'; return; } info.innerHTML = `

${flights.length} flight(s) found

`; list.innerHTML = flights.map(f => `
${f.from_code} ${f.from_city} ${f.departure_time.split(' ')[1]}
${f.duration}
✈
${f.flight_number}
${f.to_code} ${f.to_city} ${f.arrival_time.split(' ')[1]}
${f.aircraft_model} ${f.available_seats} seats left
PKR ${Number(f.price).toLocaleString()}
`).join(''); }) .catch(() => { info.innerHTML = '

Connection error. Make sure XAMPP is running.

'; }); } function searchFlights() { const from = el('from') ? el('from').value.trim().toUpperCase() : ''; const to = el('to') ? el('to').value.trim().toUpperCase() : ''; const date = el('date') ? el('date').value : ''; const cls = el('class') ? el('class').value : 'Economy'; if (!from || !to || !date) { alert('Please fill in From, To, and Date!'); return; } window.location.href = `flights.html?from=${from}&to=${to}&date=${date}&class=${cls}`; } function bookFlight(flightId, cls) { window.location.href = `booking.html?flight_id=${flightId}&class=${cls}`; }
